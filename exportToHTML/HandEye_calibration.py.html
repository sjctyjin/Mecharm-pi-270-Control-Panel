23   ''' 
24      file           : 校正影像及pos手臂資訊，儲存資料夾路徑 
25      chess_board_x_num img  : 標定板x軸點數量 預設8    
26      chess_board_y_num      : 標定板x軸點數量 預設6    
27      save_RT                : 儲存手眼標定結果 預設0 不儲存，1 儲存 
28      show_RT_validate       : 顯示手眼標定驗證結果 預設1 顯示，0 顯示 
29       
30   '''
31   def HandEye_Calibration(file,camera_matrix,chess_board_x_num=8,chess_board_y_num=6,save_RT=0,show_RT_validate=1):
32       #機械手臂位置及相機拍攝圖片
33       lsdir = file
34       square_size = 1.2 # 標定板方格大小（单位：cm）
35       Cam_point = []#存放標定板原點相對於相機的座標
36       #讀取相機內參
37       with open(camera_matrix, 'r') as f:
38           data = yaml.load(f.read(),Loader = yaml.FullLoader)
39   
40       #相機內參矩陣
41       mtx  = np.array(data["camera_matrix"], dtype=np.float64)
42       dist = None
43   
44       #取得棋盤格世界座標及投影座標
45       objp,axis = define_chess(chess_board_x_num,chess_board_y_num,square_size)
46   
47       good_picture=[]#存放棋盤格圖片
48       pic_num = 1
49       for i in os.listdir(f'{lsdir}/img'):
50           if i != None:
51               if i.split('.')[1] == "jpg":
52                   # print(i)
53                   good_picture.append(pic_num)
54                   pic_num += 1
55   
56       # print(good_picture)
57       folder = f'{lsdir}/img'
58   
59       """ 
60           獲取每一張標定板到相機的變換矩陣 
61       """
62       #計算board to cam 變換矩陣
63       R_all_chess_to_cam_1=[]
64       T_all_chess_to_cam_1=[]
65   
66       #找出標定板相對於相機座標的變換矩陣，並依序存放至陣列中
67       for i in good_picture:
68   
69           image_path=folder+'/'+str(i)+'.jpg'
70           RT=get_RT_from_chessboard(image_path, chess_board_x_num, chess_board_y_num, mtx,0,objp,axis,dist,Cam_point,mtx)
71           R_all_chess_to_cam_1.append(RT[:3,:3])
72           T_all_chess_to_cam_1.append(RT[:3, 3].reshape((3,1)))
73   
74       """ 
75           獲取每一張相片拍攝時對應的手臂座標姿態變換矩陣 
76       """
77       file_address=f'{lsdir}/pos/pos.csv'#从记录文件读取机器人六个位姿
78       sheet_1 = pd.read_csv(file_address)
79   
80       #計算end to base 變換矩陣
81       R_all_end_to_base_1=[]
82       T_all_end_to_base_1=[]
83   
84       #找出相機拍攝照片時對應的手臂姿態位置，並依序存放至陣列中
85       for i in range(len(good_picture)):
86   
87           RT=pose_robot(sheet_1.iloc[i]['ax'],sheet_1.iloc[i]['ay'],sheet_1.iloc[i]['az'],sheet_1.iloc[i]['dx'],
88                                             sheet_1.iloc[i]['dy'],sheet_1.iloc[i]['dz'])
89   
90           R_all_end_to_base_1.append(RT[:3, :3])
91           T_all_end_to_base_1.append(RT[:3, 3].reshape((3, 1)))
92   
93       # print("末端到基座旋轉矩陣 : ",R_all_end_to_base_1)
94       # print("末端到基座平移向量 : ",T_all_end_to_base_1)
95       # print("標定板到相機旋轉矩陣 : ",R_all_chess_to_cam_1)
96       # print("標定板到相機平移向量 : ",T_all_chess_to_cam_1)
97   
98       #手眼標定函數，method=3代表使用CALIB_HAND_EYE_ANDREFF 標定方法
99       R,T=cv2.calibrateHandEye(R_all_end_to_base_1,T_all_end_to_base_1,R_all_chess_to_cam_1,T_all_chess_to_cam_1,method=3)#,method=cv2.CALIB_HAND_EYE_ANDREFF)#手眼标定
100  
101      print(f'相機相對於末端的旋轉矩陣 : \n{R}\n相機相對於末端的位移向量 : \n{T}')
102  
103  
104      #寫入相機與手臂末端的變換矩陣
105      if save_RT:
106          data= {'camera_to_end': np.asarray(RT).tolist()}
107          with open(f"{lsdir}/camera_to_end_{time.strftime('%Y%m%d_%H%M%S')}.yaml", "w")as f:
108              yaml.dump(data, f)
109  
110      # 根據每張圖片，與對應的手臂座標姿態，做驗證
111      for i in range(len(good_picture)):
112          vali = show_RT_validate
113          print('=' * 30)
114          print('第', i+1, '張校正板圖片')
115          print('=' * 30)
116  
117  
118          ''' Step 1. 標定板相對於相機的變換矩陣 OK'''
119  
120          RT_chess_to_cam=np.column_stack((R_all_chess_to_cam_1[i],T_all_chess_to_cam_1[i]))
121          RT_chess_to_cam=np.row_stack((RT_chess_to_cam,np.array([0,0,0,1])))
122          print("1.驗證標定板相對於相機的變換矩陣 : ",np.dot(RT_chess_to_cam, [0,0,0,1])) if vali else print('')
123          print('=' * 200)
124  
125          '''==============================================================================='''
126  
127  
128          '''Step 2.相機相對於末端的變換矩陣'''
129  
130          RT_cam_to_end=np.column_stack((R,T))
131          RT_cam_to_end=np.row_stack((RT_cam_to_end,np.array([0,0,0,1])))
132          print("2.驗證相機相對於手臂末端的變換矩陣 = (相機與末端的安裝距離):　", np.dot(RT_cam_to_end, [0, 0, 0, 1])) if vali else print('')# 是相機到末端
133          print('=' * 200)
134  
135          '''==============================================================================='''
136  
137          ''' Step 3.末端到基座的變換矩陣 OK'''
138  
139          RT_end_to_base = np.column_stack((R_all_end_to_base_1[i], T_all_end_to_base_1[i]))
140          RT_end_to_base = np.row_stack((RT_end_to_base, np.array([0, 0, 0, 1])))
141          # 將變換矩陣轉變為旋轉矩陣
142  
143          print("3.末端相對於基座的變換矩陣 : \n",RT_end_to_base)
144          print('=' * 200)
145  
146          thedaX = atan2(RT_end_to_base[2, 1], RT_end_to_base[2, 2]) * 180 / np.pi
147          thedaY = atan2(RT_end_to_base[2, 0] * -1, sqrt(RT_end_to_base[2, 0] ** 2 + RT_end_to_base[2, 2] ** 2)) * 180 / np.pi
148          thedaZ = atan2(RT_end_to_base[1, 0], RT_end_to_base[0, 0]) * 180 / np.pi
149          print('3.1 驗證末端相對於基座的變換矩陣 = (當前手臂座標) ',np.dot(RT_end_to_base, [0,0,0,1])) if vali else print('')
150          print("3.2 驗證手臂末端姿態 :　",(thedaX,thedaY,thedaZ))
151          print(f"3.3 實際手臂座標 : \nX:{sheet_1.iloc[i]['dx']},Y:{sheet_1.iloc[i]['dy']},Z:{sheet_1.iloc[i]['dz']}\nrx:{sheet_1.iloc[i]['ax']},ry:{sheet_1.iloc[i]['ay']},rz:{sheet_1.iloc[i]['az']}")
152          print('=' * 200)
153  
154          '''==============================================================================='''
155  
156          ''' 標定板 to 相機 to 末端 to 基座 '''
157          RT_chess_to_base=RT_end_to_base@RT_cam_to_end#即为固定的棋盘格相对于机器人基坐标系位姿
158          image_path = folder + '/' + str(i+1) + '.jpg'
159          coord = [Cam_point[i][0][0]*10, Cam_point[i][1][0]*10, Cam_point[i][2][0]*10,1]  # 座標，待測
160          print("4.標定板原點相對於相機座標 (相機原點與標定板距離): ",coord) if vali else print('')
161          print('=' * 200)
162          print("5.標定板相對於末端座標 : ",np.dot(RT, coord)) if vali else print('')
163          print('=' * 200)
164          print("6.末端轉換手臂基座座標為 : ", (RT_end_to_base@RT_cam_to_end)@coord) if vali else print('')
165          print('=' * 200)
166          print("7.相機座標相對於機械手臂基座的變換矩陣 : \n",RT_chess_to_base)
167          print('*'*200)
168          # 畫出棋盤格投影座標
169          get_RT_from_chessboard(image_path, chess_board_x_num, chess_board_y_num, mtx, 1, objp, axis, dist, Cam_point, mtx)
