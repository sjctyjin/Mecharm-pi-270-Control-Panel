13   def define_chess(chess_board_x_num,chess_board_y_num,square_size):
14       ''' 
15           :param chess_board_x_num: 棋盤格x方向格子數 
16           :param chess_board_y_num: 棋盤格y方向格子數 
17           :param square_size: 單位棋盤格長度,cm 
18           :return: 棋盤格世界座標，投影坐標系 
19       '''
20       #定義棋盤格世界座標
21       objp = np.zeros((chess_board_x_num*chess_board_y_num,3), np.float32)
22       objp[:,:2] = np.mgrid[0:chess_board_x_num,0:chess_board_y_num].T.reshape(-1,2)*square_size
23   
24       #繪製出棋盤格投影座標-長度為3格，Z軸設定為-3即出紙面 為-Z
25       axis = np.float32([[3,0,0], [0,3,0], [0,0,-3]]).reshape(-1,3)*square_size
26       return objp,axis

29   #畫出投影坐標系
30   def draw(img, corners, imgpts,tvecs):
31       ''' 
32           :param img: 圖片 
33           :param corners: 棋盤格上的每個角點 
34           :param imgpts: projectPoints求得的投影座標 
35           :param tvecs: solvepnp求得的位移向量 
36           :return: img : 畫上投影座標後的圖片 
37       '''
38       corner = tuple(map(int, corners[0].ravel()))
39       img = cv2.line(img, corner, tuple(map(int,imgpts[0].ravel())), (0,0,255), 5)#X
40       img = cv2.line(img, corner, tuple(map(int,imgpts[1].ravel())), (0,255,0), 5)#Y
41       img = cv2.line(img, corner, tuple(map(int,imgpts[2].ravel())), (255,0,0), 5)#Z
42       x, y = corner
43       cv2.putText(img, f"(X:{round(tvecs[0][0], 2)*10}mm,Y:{round(tvecs[1][0], 2)*10}mm,Z:{round(tvecs[2][0], 2)*10}mm)",
44                   (x,y-45), cv2.FONT_HERSHEY_SIMPLEX,
45                   1, (55, 200, 255), 2, cv2.LINE_AA)
46       return img

49   #根據歐拉角計算旋轉矩陣
50   def myRPY2R_robot(x, y, z):
51       ''' 
52           :param x: 手臂歐拉角 rx (弧度) 
53           :param y: 手臂歐拉角 ry (弧度) 
54           :param z: 手臂歐拉角 rz (弧度) 
55           :return: R : 手臂末端相對於手臂基座的旋轉矩陣 
56       '''
57       Rx = np.array([[1, 0, 0], [0, cos(x), -sin(x)], [0, sin(x), cos(x)]])
58       Ry = np.array([[cos(y), 0, sin(y)], [0, 1, 0], [-sin(y), 0, cos(y)]])
59       Rz = np.array([[cos(z), -sin(z), 0], [sin(z), cos(z), 0], [0, 0, 1]])
60       R = Rz @ Ry @ Rx
61       return R

63   #根據手臂位置姿態計算變換矩陣
64   def pose_robot(x, y, z, Tx, Ty, Tz):
65       ''' 
66           :param x: 手臂歐拉角 rx (弧度) 
67           :param y: 手臂歐拉角 ry (弧度) 
68           :param z: 手臂歐拉角 rz (弧度) 
69           :param Tx: 手臂X座標 
70           :param Ty: 手臂Y座標 
71           :param Tz: 手臂Z座標 
72           :return: RT1 : 手臂末端相對於手臂基座的變換矩陣 
73       '''
74       thetaX = x / 180 * pi #將角度轉弧度
75       thetaY = y / 180 * pi
76       thetaZ = z / 180 * pi
77       R = myRPY2R_robot(thetaX, thetaY, thetaZ)#計算旋轉矩陣
78       t = np.array([[Tx], [Ty], [Tz]])#手臂座標 = 末端位移向量
79       RT1 = np.column_stack([R, t])  # 列合并
80       RT1 = np.row_stack((RT1, np.array([0,0,0,1])))
81       return RT1

83   #取得相機外參: 標定板座標原點相對於相機原點的變換矩陣
84   def get_RT_from_chessboard(img_path,chess_board_x_num,chess_board_y_num,K,check,objp,axis,dist,Cam_point,mtx):
85       ''' 
86       :param img_path: 讀取圖片 
87       :param chess_board_x_num: 棋盤格x方向格子數 
88       :param chess_board_y_num: 棋盤格y方向格子數 
89       :param K: 相機內參 
90       :param check: 是否停下查看圖片,0 = 不停下快速讀完,1 = 停下查看投影圖片 
91       :param objp: 棋盤格世界座標 
92       :param axis: 投影座標 
93       :param dist: 畸變系數 
94       :param Cam_point: 存放每張棋盤個原點solvepnp後得到的位移向量 
95       :param mtx: 相機內參 
96    
97       :return: RT 相機外參 
98       '''
99   
100      img = cv2.imread(img_path)
101      gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
102      ret, corners = cv2.findChessboardCorners(gray, (chess_board_x_num, chess_board_y_num), None)
103      criteria = (cv2.TERM_CRITERIA_EPS + cv2.TERM_CRITERIA_MAX_ITER, 30, 0.001)
104      if ret == True:
105          try:
106              corners2 = cv2.cornerSubPix(gray, corners, (11, 11), (-1, -1), criteria)
107  
108              # Find the rotation and translation vectors.
109              ret, rvecs, tvecs = cv2.solvePnP(objp, corners2, K, dist)
110              #平移向量 = 標定板原點相對於相機座標系的座標值
111              # print("旋轉向量 : ", rvecs)
112              # print("平移向量 : ", tvecs)
113              # print("旋轉角度 : ",np.degrees(rvecs))
114              Cam_point.append(tvecs)#獲取每張圖的棋盤格原點
115              # project 3D points to image plane
116              imgpts, jac = cv2.projectPoints(axis, rvecs, tvecs, mtx, dist)
117  
118              img = draw(img, corners2, imgpts,tvecs)
119              img = cv2.circle(img, (img.shape[1] // 2, img.shape[0] // 2), 2, (255, 0, 0), -1)
120  
121          except:
122              print("wrong")
123              print(traceback.print_exc())
124  
125      RT=np.column_stack(((cv2.Rodrigues(rvecs))[0],tvecs))
126      RT = np.row_stack((RT, np.array([0, 0, 0, 1])))
127      img = draw(img, corners2, imgpts,tvecs)
128      cv2.imshow('img', img)
129      #check = 1 停下查看棋盤格投影座標圖片, check = 0 快速完成
130      if check:
131          cv2.waitKey(0)
132      else:
133          cv2.waitKey(1)
134  
135      return RT
